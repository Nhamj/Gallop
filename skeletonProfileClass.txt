	<script>
		// Create the profile object
		var profile = {
  			pn : undefined,
  			un : undefined,
  			pw : undefined
		};
        
        // Retrieve input
        function getInput() {
        	profile.pn = document.getElementById("firstName").value +
        	" " + document.getElementById("lastName").value;
        	profile.un = document.getElementById("username").value;
        	profile.pw = document.getElementById("password").value;
		document.write("Account Created!");
        }
        
        // Log-in function
        // TODO: have it search a set of users for matching username
        // and check if password matches
        function login(username, password) {
        	if(username == profile.un && password == profile.pw)
            	return true;
            else
            	return false;
        }
